\section{Data Synchronization Flow}

\begin{lstlisting}[language=Kotlin, caption=Firebase Sync Example, label=lst:firebase_sync]
// Listen to device state changes
FirebaseDatabase.getInstance()
    .getReference("device_states/$deviceId/relay_1")
    .addValueEventListener(object : ValueEventListener {
        override fun onDataChange(snapshot: DataSnapshot) {
            val state = snapshot.child("state").value as? String
            val lastToggled = snapshot.child("lastToggled")
                .value as? Long
            
            // Update Room database
            roomDatabase.relayDao()
                .updateRelay(deviceId, 1, state, lastToggled)
        }
        
        override fun onCancelled(error: DatabaseError) {
            Log.e("Firebase", "Sync cancelled: ${error.message}")
        }
    })

// Write user command
fun toggleRelay(deviceId: String, relay: Int) {
    val command = mapOf(
        "relay" to relay,
        "action" to "ON",
        "source" to "user",
        "timestamp" to System.currentTimeMillis(),
        "executed" to false
    )
    
    FirebaseDatabase.getInstance()
        .getReference("commands/$deviceId/${System.currentTimeMillis()}")
        .setValue(command)
        .addOnSuccessListener {
            Log.d("Firebase", "Command sent successfully")
        }
        .addOnFailureListener { error ->
            Log.e("Firebase", "Command failed: ${error.message}")
            // Add to Room queue for retry
            roomDatabase.commandQueueDao().insert(
                QueuedCommand(deviceId, relay, "ON", System.currentTimeMillis())
            )
        }
}
\end{lstlisting}

\section{Database Size Estimation}

For a typical household with 4 devices, 20 schedules, 5 scenes, and 3 months of logs:

\begin{itemize}
    \item Devices metadata: ~4 KB
    \item Device states (real-time): ~5 KB
    \item Commands (transient, cleaned up): ~1 MB historical
    \item Schedules: ~15 KB
    \item Scenes: ~8 KB
    \item Automation rules: ~12 KB
    \item Presence data: ~2 KB
    \item Billing cycles (3 months): ~10 KB
    \item Logs (3 months): ~5 MB
    \item \textbf{Total Estimated:} ~6.1 MB (well within Firebase free tier of 1 GB)
\end{itemize}

This schema ensures efficient storage, fast queries, and appropriate data isolation for security.
